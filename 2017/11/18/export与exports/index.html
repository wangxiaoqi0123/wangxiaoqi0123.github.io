export与exports
发表于 2017-11-18| 更新于 2019-12-20|javascript
阅读量:

ES5 方式

ES5 解构导出

exports.key1 = value1
exports.key2 = value2
exports.key3 = value3

var {key1,key2,key3} = require(‘路径’)

ES5 对象导出

module.exports = { …}

var obj = require(‘路径’)

ES6 方式

ES6 结构导出

export const key = value
export function fn () {}
export class Fn{}

import {key,fn,Fn} from “路径”

ES6 对象导出

export default {…}

import obj from “路径”

ES6 模块导入导出 只适合 结构的方式
export { key1 , key2} from “路径”
